1. Class Diagram (User -> Customer, PaymentMethod -> CardPayment)

classDiagram
    class User {
        <<abstract>>
        -int id
        -String username
        -String email
    }

    class Customer {
        -String address
        -String phone
    }

    class PaymentMethod {
        <<interface>>
        +processPayment(double amount) boolean
        +refundPayment(double amount) void
    }

    class CardPayment {
        -String cardNumber
        -String cardholderName
        -double balance
    }

    class CODPayment {
        -double cashHanded
    }

    User <|-- Customer
    User <|-- RestaurantOwner
    PaymentMethod <|.. CardPayment
    PaymentMethod <|.. CODPayment

    MainServer --> SystemFacade
    SystemFacade --> Store
    Store "1" *-- "*" Item : contains
    Store "1" *-- "*" OrderCart : manages
    Item <|-- Meal
    Item <|-- BeverageItem
    OrderCart "1" *-- "1" Customer : belongs to
    OrderCart "1" o-- "*" Item : includes
    OrderCart --> PaymentMethod : uses

2. Database Entity-Relationship (ER) Diagram

erDiagram
    USERS ||--o{ ORDERS : places
    ORDERS ||--|{ ORDER_ITEMS : contains
    MENU_ITEMS ||--o{ ORDER_ITEMS : "included in"

    USERS {
        int id PK
        string username
        string password
        string email
        string role
    }

    MENU_ITEMS {
        int id PK
        string name
        string description
        decimal price
        string category
    }

    ORDERS {
        string order_id PK
        int user_id FK
        decimal total_amount
        datetime created_at
        string status
    }

    ORDER_ITEMS {
        int id PK
        string order_id FK
        int menu_item_id FK
        int quantity
        decimal unit_price
    }
